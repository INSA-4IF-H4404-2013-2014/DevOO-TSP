    one node already

Superviseur->+ ":MainWindow": clic on "supprimer" button
":MainWindow" ->+ ":DeleteController": actionPerformed(ActionEvent e)

":DeleteController" ->+ ":Round" : isPartOf(Delivery d)
opt selected delivery is a part of the round
	":Round" --> ":DeleteController" : false
else 
	":Round" -->- ":DeleteController" : true
	":DeleteController" ->+ ":RemoveDelivery" : new(MainWindowController controller, Delivery d)
	":RemoveDelivery" -->- ":DeleteController" :
	":DeleteController" ->+ ":RemoveDelivery" : apply
	":RemoveDelivery" -->- ":DeleteController" :
end

":DeleteController" -->- ":MainWindow" :
":MainWindow" -->- Superviseur : repaint
